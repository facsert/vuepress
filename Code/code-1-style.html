                  # 突出重点

                  通过 前缀 后缀 突出重点

                  前缀含义
                  is_xx是不是
                  has_xx有没有
                  should_xx是否应该
                  can_xx能不能

                  避免反义命名

                   test_pass = True                                # 推荐 True 一般与成功, 正常对应 
 test_fail = False                               # 不推荐 False 一般与错误, 失败对应

                  度量命名带上单位

                   time_secs = 8
 time_hours = 2
 time_days = 3
