def average(array):
    array_sum, array_length = 0, len(array)
    for i in array:
        array_sum += i
    return array_sum / array_length

def compare_list_average(list_a, list_b):
    a_average = average(list_a)
    b_average = average(list_b)
    return a_average > a_average
