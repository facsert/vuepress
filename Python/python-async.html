func Sum(name string, end int) {
    defer wg.Done()
    fmt.Println(name, end, time.Now())
    sum := 0
    for i := 0; i <= end; i++ {
        sum += i
    }
    fmt.Println(name, end, time.Now())
}

func main() {
    names := [...]string{"first", "second", "third"}
    delays := [...]int{15000000000, 1000000000, 13000000000}
    for i := 0; i < 3; i++ {
        wg.Add(1)
        go Sum(names[i], delays[i])
    }
    fmt.Println("Over")
    wg.Wait()
}
